║                       COMANDOS MAIS IMPORTANTES                           ║
╚════════════════════════════════════════════════════════════════════════════╝

INICIAR:
  docker-compose up -d

VERIFICAR STATUS:
  docker-compose ps

VER LOGS:
  docker-compose logs -f

PARAR:
  docker-compose down

ENTRAR NO CONTAINER:
  docker-compose exec app sh

BANCO DE DADOS:
  docker-compose exec db mysql -uroot -proot -Dbiblioteca
